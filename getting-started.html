Getting started

Please note: this page has been marked as incomplete.

You'll need a to use the same basic core template for any page where you want to use the framework.

First, download the latest source code. There are two folders: gui and js containing all the assets. Next, copy the core HTML template and update the paths to the gui and js assets to reflect your directory structure:

<!DOCTYPE html>
<html lang="en">
<head>
<meta charset="utf-8" />
<title>Page title - Site title - Open University</title>

<link rel="stylesheet" type="text/css" href="gui/styles.css" media="screen, projection" />
<link rel="stylesheet" type="text/css" href="gui/header-footer.css" media="screen, projection" />

<!-- optional font awesome -->
<link rel="stylesheet" type="text/css" href="gui/fonts/font-awesome/font-awesome.min.css" media="screen, projection" />

<meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no" />

<!-- favicons -->
<link rel="apple-touch-icon" href="gui/icons/apple-touch-icon.png">
<link rel="icon" href="gui/icons/favicon.png">
</head>

<body class="">

<div id="ou-org">
<a class="ou-skip" href="#ou-content"><i class="fa fa-arrow-circle-down"></i> Skip to content</a>
<div id="ou-header">
<!-- insert OU header here -->
</div> <!-- end ou-header -->

<div id="ou-site">

<div id="ou-site-header">
<!-- insert optional site-specific header here -->
</div>
<!-- end ou-site-header -->

<div id="ou-site-body">
<div id="ou-page">

<div id="ou-region0">
<ol class="ou-ancestors">
<li><a href="index.html">Home</a></li>
<li>Current page</li>
</ol>
</div> <!-- end ou-region0 -->

<div id="ou-region1">
<div id="ou-content" class="ou-content">
<h1>Page title</h1>
<p>Page content.</p>
</div> <!-- end ou-content -->
</div> <!-- end ou-region1 -->

<div id="ou-region2">
</div> <!-- end ou-region2 -->

</div> <!-- end ou-page -->
</div> <!-- end ou-site-body -->

<div id="ou-site-footer">
<a class="ou-to-top" href="#ou-content"><i class="fa fa-arrow-circle-up"></i> Back to top</a>
</div>
<!-- end ou-site-footer -->

</div> <!-- end ou-site -->

<div id="ou-footer">
<!-- insert relevant OU footer here -->
</div> <!-- end ou-footer -->

</div> <!-- end ou-org -->

<!-- Javascripts -->
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.11.1/jquery.min.js"></script>
<script src="js/scripts.js"></script>


</body>
</html>

You are now free to start adding other components.
